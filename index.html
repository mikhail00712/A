function parseFlightTimes(text) { const timeRegex = /(\d+)\s+(\d+)\s+(\d+)\s+мин/g; let match; let totalMinutes = 0; while ((match = timeRegex.exec(text)) !== null) { const hours = parseInt(match[1]); const minutes = parseInt(match[2]); const extraMinutes = parseInt(match[3]); if (!isNaN(hours) && !isNaN(minutes) && !isNaN(extraMinutes)) { totalMinutes += hours * 60 + minutes + extraMinutes; } } return totalMinutes; }
